\begin{columns}

\column{0.35\textwidth}
\footnotesize
\begin{itemize}
  \item \textbf{Neuronale Netze} sind außerhalb der Fachwelt die
        bekanntesten Lernverfahren.
  \item Lange wurden sie kaum verwendet wegen ihrer starken Tendenz zur
        \textbf{Überanpassung}.
  \item Neuere Entwicklungen und aktuelle \textbf{Hardware} erlauben heute

  \item Vorbild ist das \textbf{menschliche Gehirn} aus vernetzten
        \textbf{Neuronen}.
\end{itemize}

\begin{frame}{Neuronale Netze: Ursprung}
\begin{columns}
\column{0.60\textwidth}
\begin{itemize}
  \item Die Anfänge liegen bei \textbf{McCulloch und Pitts} in den 1940er
        Jahren.
  \item Ziel: einfaches \textbf{mathematisches Modell} für das
        Verhalten der Neuronen im Gehirn.
  \item In Medien werden Netze oft mit einem \textbf{künstlichen Gehirn} gleichgesetzt --
        das ist irreführend.
  \item Das Gehirn ist als Ganzes \textbf{nicht verstanden}; Netze lassen
        sich daher nicht über das Gehirn erklären.
  \item Hilfreicher ist der Blick auf die \textbf{Verkettung von
        Funktionen}.
\end{itemize}

\column{0.48\textwidth}
\begin{itemize}
  \item Elementarer Baustein ist das \textbf{künstliche Neuron}.
  \item Links stehen die Einträge $x_0, x_1, \dots, x_n$ des
        \textbf{Merkmalsvektors} $\mathbf{x}$.
  \item Sie werden mit \textbf{Gewichten} $w_i \in \mathbb{R}$ multipliziert
        und aufsummiert.
  \item Das Ergebnis $h$ geht in die \textbf{Aktivierungsfunktion} $g$, die
        die Ausgabe $y$ liefert.
\end{itemize}
\end{columns}

\end{frame}

\begin{frame}{Einlagiges Perzeptron: Summation und Analogie}
\begin{columns}
\column{0.60\textwidth}
\begin{itemize}
  \item Die gewichtete Summe ergibt die Funktion
        $h: \mathbb{R}^n \to \mathbb{R}$:
        \[
          h(\mathbf{x}) = \sum_{i=0}^{n} w_i x_i
        \]
  \item $h(\mathbf{x})$ ist der Eingangswert der \textbf{Aktivierungs\-%
        funktion} $g$.
  \item \textbf{Biologische Analogie:} Eine Nervenzelle summiert die
        Anregungen anderer Zellen.
  \item Übersteigt die Summe ein \textbf{Schwellenpotential}, \textbf{feuert}
        die Zelle.
\end{itemize}

\column{0.35\textwidth}
\begin{blockC1}{Vereinfachung}
Das Modell ist stark vereinfacht -- es ist nur eine \textbf{Anlehnung} an
die echte Nervenzelle.
\end{blockC1}
\end{columns}
\source{\cite{Frochte2020}}
\end{frame}

\begin{frame}{Einlagiges Perzeptron: Schwellwertfunktion}
\begin{columns}
\column{0.47\textwidth}
\begin{itemize}
  \item Die mathematische Nachbildung \textbf{feuert} ab Überschreitung
        eines \textbf{Schwellenwertes} $\theta$.
  \item Sie springt dabei von $0$ auf $1$ (\textbf{Sprungfunktion}).
  \item Unterhalb von $\theta$ ist das Neuron \textbf{inaktiv}:
        \[
          y = g(h) =
          \begin{cases}
            1 & \text{für } h > \theta\\
            0 & \text{für } h \leq \theta
          \end{cases}
        \]
\end{itemize}

\begin{frame}{Einlagiges Perzeptron: Kompaktere Darstellung}
    \small
\begin{columns}
\column{0.60\textwidth}
\begin{itemize}
  \item Das Neuron lässt sich \textbf{mathematisch kompakt} darstellen.
  \item Jeder Eintrag $x_i$ wird mit dem Gewicht $w_i$ multipliziert und
        aufsummiert:
        \[
          h(\mathbf{x}) = \sum_{i=0}^{n} w_i x_i
        \]
  \item Fasst man Gewichte und Eingabe als \textbf{Vektoren}
        $\mathbf{w} = (w_0, \dots, w_n)^T$ und
        $\mathbf{x} = (x_0, \dots, x_n)^T$ auf, wird daraus ein
        \textbf{Skalarprodukt}:
        \[
          h(\mathbf{x}) = \mathbf{w}\cdot\mathbf{x}
        \]
  \item Die Ausgabe ist dann $y = g(h(\mathbf{x})) = g(\mathbf{w}\cdot\mathbf{x})$.
\end{itemize}

\column{0.35\textwidth}
\begin{blockC1}{\small Ziel}
Eine \textbf{knappe Notation}, die Gewichtung, Summation und Aktivierung in
einem Ausdruck zusammenfasst.
\end{blockC1}
\end{columns}
\source{\cite{Frochte2020}}
\end{frame}


\begin{frame}{Perzeptron: Schwellwert als Trenngerade}
\small
\begin{columns}
\column{0.47\textwidth}
\begin{itemize}
  \item Wir nehmen jetzt noch den den Schwellwert $\theta$ hinzu und stellen um:
        \[
          h(\mathbf{x}) - \theta = \mathbf{w}\cdot\mathbf{x} - \theta = 0
        \]
  \item Das ist eine \textbf{Hyperebenen-Gleichung}, wie wir sie von den
        \textbf{SVMs} kennen.
  \item $\theta$ verschiebt die Hyperebene aus dem \textbf{Ursprung}
        heraus.
  \item Im 2D-Fall ist die Hyperebene eine \textbf{Gerade}:
        \[
          w_0 x_0 + w_1 x_1 - \theta = 0
        \]
\end{itemize}

\begin{frame}{Perzeptron: Matrix-Vektor-Darstellung}
\footnotesize
\begin{columns}
\column{0.60\textwidth}
\begin{itemize}
  \item Die Gewichte werden so nummeriert, dass die Summation eine
        \textbf{Matrix-Vektor-Multiplikation} wird. Mit
        \[
          W = \begin{pmatrix*}[r]
            w_{1,1} & w_{1,2} & w_{1,3}\\
            w_{2,1} & w_{2,2} & w_{2,3}
          \end{pmatrix*}
        \]
        gilt $\mathbf{h} = W\cdot\mathbf{x}$ und
        $\mathbf{y} = g(\mathbf{h}(\mathbf{x}))$.
  \item $g$ wird \textbf{komponentenweise} angewandt (Schwellwert-/%
        Heaviside-Funktion).
  \item Am Beispiel $y_1$ zeigt sich die Rolle des Bias-Neurons:
        \[
          h(\mathbf{x}) = w_{1,1}\cdot x_1 + w_{1,2}\cdot x_2 + w_{1,3}\cdot 1
        \]
  \item $w_{1,3}$ übernimmt die Rolle von $-\theta$.
\end{itemize}

\column{0.35\textwidth}
\begin{blockC1}{\footnotesize Ein Output-Neuron}
Beim Startbeispiel genügt \textbf{ein} Output. $W$ wird einzeilig und lässt
sich als \textbf{Zeilen-Vektor} schreiben:
\[
  \mathbf{w} = \begin{pmatrix} w_1 & w_2 & w_3 \end{pmatrix}
\]
\end{blockC1}
\end{columns}
\source{\cite{Frochte2020}}
\end{frame}

\begin{frame}{Perzeptron: Hebbsche Lernregel}
\begin{columns}
\column{0.60\textwidth}
\begin{itemize}
  \item Neuronale Netze werden trainiert, indem man die \textbf{Gewichte} $w_i$ anpasst.
  \item Grundlage ist der \textbf{Fehler} zwischen gewünschtem $y$ und
        beobachtetem $\hat{y}$:
        \[
          \text{error} = y - \hat{y}
        \]
  \item Ist $y > \hat{y}$, werden die Gewichte \textbf{vergrößert}, sonst
        \textbf{verkleinert}:
        \[
          \Delta w_j = \text{error}\cdot x_j, \qquad w_j = w_j + \Delta w_j
        \]
  \item Eine \textbf{Lernrate} $\eta \in\, ]0,1]$ \textbf{dämpft} die
        Änderung und sichert die Konvergenz.
\end{itemize}

\column{0.35\textwidth}
\begin{blockC1}{Grundidee}
Gewichte mit \textbf{großem Einfluss} werden stärker verändert als solche
mit kleinem Einfluss.
\end{blockC1}
\end{columns}
\source{\cite{Frochte2020}}
\end{frame}

\begin{frame}{Perzeptron: Algorithmus der Hebbschen Lernregel}
\footnotesize
\begin{columns}
\SetAlgoCaptionLayout{footnotesize}
\column{0.60\textwidth}
\begin{algorithm}[H]
\SetAlgoLined
\KwIn{Beispielmenge $D = \{(x, y)\}$}
\KwOut{Gewichtsvektor $\mathbf{w}$}
$\mathbf{w}$ mit Zufallszahlen in $[-0.5, 0.5]$ initialisieren\;
$t = 0$\;
\Repeat{$\lVert Y - \text{heaviside}(WX)\rVert < \varepsilon$ \textbf{oder} $t > t_{\max}$}{
  $t = t + 1$\;
  $(x, y) = \text{RandomSelect}(D)$\;
  $\hat{y} = \text{heaviside}(W \cdot x)$\;
  $\text{error} = y - \hat{y}$\;
  \For{$j = 0$ \KwTo $p$}{
    $\Delta w_j = \eta\cdot\text{error}\cdot x_j$\;
    $w_j = w_j + \Delta w_j$\;
  }
}
\caption{Hebbsche Lernregel}
\end{algorithm}

\column{0.35\textwidth}
\begin{blockC1}{\footnotesize Ablauf}
\begin{itemize}
  \item Zufälliges Beispiel $(x, y)$ wählen.
  \item Ausgabe berechnen:
        $\hat{y} = \text{heaviside}(W \cdot x)$.
  \item \textbf{Fehler} bestimmen:
        $\text{error} = y - \hat{y}$.
  \item Pro Gewicht die Änderung berechnen:
        $\Delta w_j = \eta\cdot\text{error}\cdot x_j$.
  \item Gewicht anpassen: $w_j = w_j + \Delta w_j$.
  \item Wiederholen, bis der Fehler $< \varepsilon$ ist oder $t_{\max}$
        erreicht ist.
\end{itemize}
\end{blockC1}
\end{columns}
\source{\cite{Frochte2020}}
\end{frame}

\begin{frame}{Hebbsche Lernregel -- Eigenschaften}
\begin{columns}
\begin{column}{0.6\textwidth}
\begin{itemize}
    \item Der Lernalgorithmus verwendet \textbf{Zufallszahlen} $\Rightarrow$ seed beeinflußt Ergebnis.
    \item Er liefert eine \textbf{Trennebene}, aber \textbf{keine optimale} im geometrischen Sinn.
    \item Qualität der Lösung variiert.
    \item Weitere Verfeinerungen möglich und notwendig.
    \item Dennoch bemerkenswert: funktionierender Lernalgorithmus \textbf{seit 1947}.
\end{itemize}
\end{column}

\makesubsubsectionpage{Multilayer Perceptron}

\begin{frame}{Was kann ein einlagiges Perzeptron wirklich?}

\begin{columns}

\column{0.55\textwidth}
\footnotesize
\begin{itemize}
    \item Ein einlagiges Perzeptron trennt Klassen exakt, wenn die Daten \textbf{linear separierbar} sind.
    \item \textbf{OR}: $(0,0)$ liefert $0$, alle anderen $1$. Der Nullpunkt lässt sich mit \textbf{einer Geraden} abtrennen $\Rightarrow$ \textbf{lernbar}.
    \item \textbf{AND}: $(1,1)$ liefert $1$, alle anderen $0$. Der Einspunkt ist mit \textbf{einer Geraden} abtrennbar $\Rightarrow$ \textbf{lernbar}.
    \item \textbf{XOR}: $(0,1)$ und $(1,0)$ liefern $1$, $(0,0)$ und $(1,1)$ liefern $0$. Die beiden Klassen liegen \textbf{über Kreuz} $\Rightarrow$ \textbf{keine Gerade} trennt sie.
    \item XOR ist daher \textbf{nicht linear separierbar}. Wir brauchen \textbf{mehrschichtige} Netze.
    \item Das \textbf{XOR}-Problem ist eines der berühmtesten Probleme der KI. 
\end{itemize}

\begin{frame}{Mehrschichtige Netze: Motivation}
\begin{columns}
\column{0.55\textwidth}
\begin{itemize}
  \item Für \textbf{XOR} lassen sich keine passenden Gewichte $w$ finden.
  \item Lösung: mehrere Neuronen \textbf{neben-} und vor allem
        \textbf{hintereinander} verschalten.
  \item Solche Netze sind \textbf{weit komplexer} zu trainieren.
  \item Das Training nutzt \textbf{mathematische Optimierung} auf Basis von
        \textbf{Ableitungen}.
  \item Problem: die \textbf{Heaviside-Funktion} ist \textbf{nicht
        differenzierbar}.
\end{itemize}

\column{0.40\textwidth}
\begin{blockC1}{Konsequenz}
Wir brauchen \textbf{differenzierbare} Aktivierungsfunktionen als Ersatz für
die Heaviside-Funktion.
\end{blockC1}
\end{columns}
\source{\cite{Frochte2020}}
\end{frame}

\begin{frame}{Mehrschichtige Netze: Aktivierungsfunktionen}
\begin{columns}
\column{0.60\textwidth}
Zwei differenzierbare Alternativen zur Heaviside-Funktion:
\[
  \varphi = \text{sig}(ax) = \frac{1}{1 + \exp(-ax)}
\]
\[
  \phi = \tanh(ax) = \frac{e^{ax} - e^{-ax}}{e^{ax} + e^{-ax}}
\]

\begin{itemize}
  \item Wertebereich: \textbf{sig} zwischen $0$ und $1$, \textbf{tanh}
        zwischen $-1$ und $1$.
  \item Der Parameter $a$ steuert die \textbf{Steilheit}: je größer $a$,
        desto näher an der \textbf{Heaviside}-Funktion.
  \item Die Graphen zeigen den Fall $a = 5$.
\end{itemize}

\begin{frame}{Mehrschichtige Netze: Ableitungen}
\begin{columns}
\column{0.55\textwidth}
Beide Funktionen haben eine \textbf{einfach strukturierte} Ableitung:
\[
  \text{sig}'(t) = \text{sig}(t)\,\bigl(1 - \text{sig}(t)\bigr)
\]
\[
  \tanh'(t) = \bigl(1 + \tanh(t)\bigr)\bigl(1 - \tanh(t)\bigr)
\]

\vspace{0.3em}
\begin{itemize}
  \item Die Ableitung lässt sich direkt aus dem \textbf{Funktionswert}
        berechnen.
  \item Das macht sie für das \textbf{Training} (Backpropagation) besonders
        effizient.
\end{itemize}

\column{0.40\textwidth}
\begin{blockC1}{Vorteil}
Kein erneutes Auswerten nötig -- der bereits berechnete \textbf{Ausgabewert}
genügt für die Ableitung.
\end{blockC1}
\end{columns}
\source{\cite{Frochte2020}}
\end{frame}

\draw[->] (b1) -- (b2) node[wlab, pos=0.4] {$1$};
\end{tikzpicture}}

\column{0.55\textwidth}
\footnotesize
\begin{itemize}
  \item Das \textbf{Bias-Neuron} im Hidden-Layer ist -- anders als im
        Input-Layer -- \textbf{optional}.
  \item Alternativ könnte man alle Merkmale (inkl. Input-Bias) mit dem
        Hidden-Bias-Neuron verbinden.
  \item Mit den Gewichten $0$, $0$ und $1$ ergibt sich stets der Eingang
        \[
          0\cdot x_1 + 0\cdot x_2 + 1\cdot 1 = 1 > 0,
        \]
        also \textbf{konstant} Heaviside $= 1$.
  \item So bildet das Netz das Bias-Neuron \textbf{selbst} nach.
\end{itemize}
\end{columns}
\source{\cite{Frochte2020}}
\end{frame}

\begin{frame}{Bias-Neuronen: Fazit}
\begin{columns}
\column{0.60\textwidth}
\begin{itemize}
  \item Ein Bias-Neuron im \textbf{Input-Layer} ist \textbf{zwingend}
        nötig- Nur so lässt sich die Trennebene aus dem Ursprung
        verschieben.
  \item In \textbf{Hidden-Layern} sind Bias-Neuronen dagegen \textbf{optional}.
  \item Fehlen sie dort, kann das Netz sie durch geeignete Gewichte
        \textbf{selbst nachbilden}.
  \item Der Preis dafür: \textbf{zusätzliche Gewichte}, die im Training
        mitbestimmt werden müssen.
\end{itemize}

\column{0.35\textwidth}
\begin{blockC1}{Kernaussage}
Ein Input-Bias ist \textbf{Pflicht}. Weitere Bias-Neuronen sind eine Frage
der \textbf{Effizienz}.
\medskip
In \textbf{Standard-Implementierungen} von Multilayer-Perzeptrons sind typischerweise Bias-Neuronen \textbf{in jeder Schicht} enthalten

\end{blockC1}
\end{columns}
\source{\cite{Frochte2020}}
\end{frame}

\begin{frame}{Multilayer Perceptron (MLP)}
\begin{columns}
\column{0.60\textwidth}
\begin{itemize}
  \item Mehrlagige Perzeptrone heißen \textbf{Multilayer Perceptrons} (MLP).
  \item Sie bestehen aus \textbf{mindestens drei} Schichten von Knoten.
  \item Oft sind sie \textbf{fully connected}:
        jedes Neuron einer Schicht ist mit allen Neuronen der nächsten
        verbunden.
  \item MLP sind \textbf{grundlegend} und für viele Anwendungen
        \textbf{leistungsstark}.
\end{itemize}

\column{0.35\textwidth}
\begin{blockDef}{Feedforward-Netze}
Ein MLP gehört zur Klasse der \textbf{Feedforward}-Netze: Jede Schicht ist
nur mit der \textbf{nächsthöheren} verbunden.
\end{blockDef}
\end{columns}
\source{\cite{Frochte2020}}
\end{frame}

\begin{frame}{MLP: Gestaltung des Netzes}
\begin{columns}
\column{0.60\textwidth}
\begin{itemize}
  \item Die \textbf{Output-Schicht} kann beliebig viele Neuronen haben --
        im Beispiel nur eines zur Vereinfachung.
  \item Zur einfacheren Rechnung fehlen im Beispiel die \textbf{Bias-Neuronen}
        der Hidden-Layer.
  \item Die Funktionsgestalt lässt sich über \textbf{mehr Layer}
        (tiefer) oder \textbf{mehr Neuronen} pro Layer (breiter) steuern.
  \item Was besser ist, hängt vom \textbf{Anwendungsfall} ab.
\end{itemize}

\column{0.35\textwidth}
\begin{block}{Historisches Problem}
Mehr als zwei Hidden-Layer waren lange selten wegen der größeren
\textbf{Herausforderung beim Training}.
\end{block}
\end{columns}
\source{\cite{Frochte2020}}
\end{frame}


\begin{frame}{MLP: Vorwärtsberechnung -- erste Schicht}
\begin{columns}
\column{0.60\textwidth}
Angenommen, das Netz ist \textbf{trainiert}, mit \textbf{Sigmoid} in den
Hidden-Layern und \textbf{linearer} Aktivierung im Output.

\vspace{0.4em}
Die erste Schicht berechnet:
\[
  O^{(1)} = \text{sig}(\underbrace{W^{(1)}\mathbf{x}}_{\mathbf{b}})
\]
mit der Gewichtsmatrix
\[
  W^{(1)} =
  \begin{pmatrix*}[r]
    w^{(1)}_{1,1} & w^{(1)}_{1,2} & w^{(1)}_{1,3}\\
    w^{(1)}_{2,1} & w^{(1)}_{2,2} & w^{(1)}_{2,3}\\
    w^{(1)}_{3,1} & w^{(1)}_{3,2} & w^{(1)}_{3,3}
  \end{pmatrix*} \in \mathbb{R}^{3\times 3}
\]

\column{0.35\textwidth}
\begin{blockC2}{Komponentenweise}
Die Sigmoid-Funktion wird pro Eintrag von $\mathbf{b}$ angewandt:
\[
  \begin{pmatrix} \text{sig}(b_1)\\ \vdots\\ \text{sig}(b_n) \end{pmatrix}
  = \text{sig}(\mathbf{b})
\]
\end{blockC2}
\end{columns}
\source{\cite{Frochte2020}}
\end{frame}

\begin{frame}{MLP: Vorwärtsberechnung -- zweite Schicht}
\begin{columns}
\column{0.60\textwidth}
Der Output $O^{(1)}$ ist der \textbf{Input} der nächsten Schicht. Er wird
mit der Matrix $W^{(2)}$ multipliziert:
\[
  O^{(2)} = \text{sig}(W^{(2)} O^{(1)})
\]
mit
\[
  W^{(2)} =
  \begin{pmatrix*}[r]
    w^{(2)}_{1,1} & w^{(2)}_{1,2} & w^{(2)}_{1,3}\\
    w^{(2)}_{2,1} & w^{(2)}_{2,2} & w^{(2)}_{2,3}\\
    w^{(2)}_{3,1} & w^{(2)}_{3,2} & w^{(2)}_{3,3}\\
    w^{(2)}_{4,1} & w^{(2)}_{4,2} & w^{(2)}_{4,3}
  \end{pmatrix*} \in \mathbb{R}^{4\times 3}
\]

\column{0.35\textwidth}
\begin{blockC1}{Prinzip}
Jede Schicht besteht aus einer \textbf{Matrix-Vektor-Multiplikation}
gefolgt von einer \textbf{komponentenweisen} Aktivierung.
\end{blockC1}
\end{columns}
\source{\cite{Frochte2020}}
\end{frame}

\begin{frame}{MLP: Vorwärtsberechnung -- Gesamtformel}
\begin{columns}
\column{0.60\textwidth}
Die Output-Schicht nutzt eine \textbf{lineare} Aktivierung:
\[
  y = W^{(3)} O^{(2)}
\]
Setzt man alle Schichten ineinander ein, ergibt sich:
\[
  y = W^{(3)}\,\text{sig}\bigl(W^{(2)}\,\text{sig}(W^{(1)}\mathbf{x})\bigr)
\]

\column{0.35\textwidth}
\begin{blockC1}{Kernaussage}
Die Vorhersage $y_p$ ist eine \textbf{Verkettung} von
Aktivierungsfunktionen und \textbf{Matrix-Vektor-Multiplikationen}.
\end{blockC1}
\end{columns}
\source{\cite{Frochte2020}}
\end{frame}


\begin{frame}{Neuronale Netze: Grundidee des Trainings}

\begin{columns}
\column{0.68\textwidth}

\begin{itemize}
    \item Zunächst wird die \textbf{Struktur} des Netzes festgelegt
          (verdeckte Schichten und Neuronen).

    \item \textbf{Gewichte} werden zufällig initialisiert und beim Training so \textbf{angepasst},
          sodass das Netz die gewünschte Ausgabe liefert.

    \item Das Training ähnelt dem Einstellen eines Mischpults mit
          vielen Reglern: Die Gewichte werden \textbf{schrittweise über
          viele Iterationen optimiert}.
    \item \textbf{Grundproblem: }  Woher wissen wir, wie genau wir die einzelnen Regeler verstellen sollen? 
\end{itemize}

\begin{frame}{Neuronale Netze: Training des Netzes}
\begin{columns}
\column{0.55\textwidth}
\small
\begin{itemize}
  \item Gewichte können nicht \textbf{isoliert} angepasst werden, da Neuronen miteinander verknüpft sind.
  \item Deshalb werden \textbf{alle Gewichte gemeinsam} optimiert.
  \item Für jede Element des Trainingsdatensatzes $D$ wird die Netzausgabe $\hat{y}$ mit dem \textbf{Soll-Wert} $y_i$ verglichen.
  \item Über alle Trainingsbeispiele werden die Fehler \textbf{quadriert und aufsummiert}.
  \item Das Ergebnis ist die \textbf{Loss Function} (\emph{Verlustfunktion} $L$).
\end{itemize}

\begin{frame}{Gradientenabstieg}

\begin{columns}
\column{0.45\textwidth}
\centering
\includesvg[width=\linewidth]{figures/gradient_2D.svg}

\vspace{0.2cm}
{\scriptsize Beispielhafte Funktion der Loss Function $L$ in Abhängigkeit von einem einzigen Gewicht.}

\column{0.55\textwidth}
\footnotesize
\begin{itemize}
    \item Der \textbf{Gradientenabstieg} ist ein Optimierungsverfahren.
    
    \item Ziel ist es, die Gewichte so zu bestimmen, dass der \textbf{Gesamtfehler} minimal wird:
    \[
      L = \sum_{(x_i,y_i)\in D}
      (\hat{y}_i-y_i)^2
    \]

    \item Dazu wird die Steigung der Loss Function ausgewertet.

    \item Die Gewichte werden schrittweise in Richtung des \textbf{stärksten Fehlerabfalls} verändert.

    \item Durch viele kleine Anpassungen nähert man sich einem \textbf{Minimum der Fehlerfunktion}.
\end{itemize}

\end{columns}

\begin{frame}{Gradient und partielle Ableitungen}

\begin{columns}
\column{0.65\textwidth}

\begin{itemize}
    \item Der \textbf{Gradient} $\nabla f$ gibt die Richtung des
          \textbf{steilsten Anstiegs} einer Funktion $f$ an.

    \item Für eine Funktion
          $f(x_1,x_2,\dots,x_n)$ ist der Gradient der Vektor aller
          partiellen Ableitungen:
          \[
          \nabla f =
          \begin{pmatrix}
          \frac{\partial f}{\partial x_1}\\
          \vdots\\
          \frac{\partial f}{\partial x_n}
          \end{pmatrix}
          \]

    \item An Extremstellen gilt $\nabla f = \mathbf{0}$.
\end{itemize}

\begin{frame}{Gradientenabstieg}
\footnotesize
\begin{columns}
\column{0.65\textwidth}

\begin{itemize}
    \item Der \textbf{Gradientenabstieg} (\textit{Gradient Descent}) ist ein Verfahren zur Minimierung einer Zielfunktion.

    \item Ziel ist es, einen Parametervektor $\theta$ zu finden, für den die Fehlerfunktion minimal wird.

    \item Vorgehen:
    \begin{enumerate}
      \footnotesize
        \item Startpunkt $\theta_0$ wählen.
        \item Gradienten $\nabla f(\theta_i)$ berechnen.
        \item Schritt entgegen der Gradientenrichtung ausführen:
        \[
            \theta_{i+1}
            =
            \theta_i
            -
            \alpha \cdot \nabla f(\theta_i)
        \]
        \item Wiederholen, bis ein Minimum erreicht ist.
    \end{enumerate}

    \item Die \textbf{Lernrate} $\alpha$ bestimmt die Schrittweite.
\end{itemize}

\column{0.35\textwidth}

\centering
\includesvg[width=\linewidth]{figures/gradient_3D_pfad.svg}

\end{columns}

\begin{frame}{Stochastic Gradient Descent}

\begin{columns}
\column{0.55\textwidth}

\begin{itemize}
    \item Beim Training sucht man Gewichte, die den \textbf{Gesamtfehler}
          minimieren:
          \[
              L =
              \sum_{(x_i,y_i)\in D}
              \left(\hat{y}_i - y_i\right)^2
          \]

    \item Bei großen Trainingsdatenmengen ist die Berechnung über alle
          Beispiele in $D$ \textbf{zeitaufwendig}.

    \item Daher wird der Gradient oft nur auf Basis eines Teils der
          Trainingsdaten berechnet.
\end{itemize}

\column{0.40\textwidth}

\begin{blockC1}{Grundidee}
\textbf{Stochastic Gradient Descent} verwendet pro Optimierungsschritt nur
einzelne Beispiele oder kleine Teilmengen der Trainingsdaten.

\vspace{0.3cm}

Dadurch wird das Training schneller, der Gradient aber nur
\textbf{angenähert}.
\end{blockC1}

\end{columns}

\source{\cite{seemann2024mlaz}}
\end{frame}

\begin{frame}{Stochastic Gradient Descent: Batches und Epochen}

\begin{columns}
\footnotesize
\column{0.40\textwidth}

\begin{itemize}
    \item Die Trainingsdatenmenge $D$ wird in kleinere Teilmengen zerlegt:
          \[
              D = B_1 \cup B_2 \cup \dots \cup B_n
          \]

    \item Diese Teilmengen heißen \textbf{Batches}.

    \item Pro Iterationsschritt wird die Kostensumme nur für einen Batch berechnet.

    \item Eine \textbf{Epoche} ist abgeschlossen, wenn alle Batches einmal
          durchlaufen wurden.
\end{itemize}

\begin{frame}{Motivation: Warum Backpropagation?}

\begin{columns}
\column{0.55\textwidth}

\begin{itemize}
    \item Beim Training werden die \textbf{Gewichte} so verändert, dass der
          Gesamtfehler kleiner wird.

    \item Für den Gradientenabstieg benötigt man den Gradienten der
          loss function:
          \[
              \nabla L
          \]

    \item Dazu muss die Loss Function nach \textbf{allen Gewichten}
          abgeleitet werden.

    \item Jede Ableitung beschreibt, wie stark ein Gewicht zum Verlust beiträgt.
\end{itemize}

\column{0.40\textwidth}

\begin{blockC1}{Ziel}
Für jedes Gewicht wird benötigt:
\[
    \frac{\partial L}{\partial w}
\]

\vspace{0.3cm}

Diese Werte bestimmen die Richtung der Gewichtsänderung.
\[
    w \leftarrow w - \alpha \frac{\partial L}{\partial w}
\]
\end{blockC1}

\end{columns}
\end{frame}

\begin{frame}{Fehlerfunktion als Funktion der Gewichte}

\begin{columns}
\column{0.52\textwidth}

\begin{itemize}
    \item Für ein Trainingsbeispiel $(x_i,y_i)\in D$ erzeugt das Netz eine
          Ausgabe $\hat{y}_i$.

    \item Die Loss Function summiert die Abweichungen über alle
          Trainingsbeispiele.

    \item Da $\hat{y}_i$ von den Gewichten abhängt, ist auch $L$ eine Funktion
          aller Gewichte.

    \item Gesucht sind Gewichte, für die $L$ minimal wird.
\end{itemize}

\column{0.45\textwidth}

\begin{block}{Gesamtfehler}
\[
    L =
    \sum_{(x_i,y_i)\in D}
    \left(\hat{y}_i-y_i\right)^2
\]
\end{block}

\vspace{0.3cm}

\begin{blockC2}{Netzausgabe (mit 2 Hidden Layer)}
\[
  \hat{y}
  =
  W^{(3)}\,\operatorname{sig}\Bigl(
  W^{(2)}\,\operatorname{sig}
  \bigl(W^{(1)}\mathbf{x}\bigr)
  \Bigr)
\]
\end{blockC2}

\end{columns}
\end{frame}

\begin{frame}{Naiver Ansatz: Direkt ableiten}
\small
\begin{columns}
\column{0.47\textwidth}

\begin{itemize}
    \item Für jedes Trainingsbeispiel müsste man $x_i$ in \textbf{die Netzfunktion
          einsetzen}.

    \item Anschließend würde man die Netzfunktion in die Fehlerfunktion
          einsetzen.

    \item Danach müssten die Richtungsableitungen bzw. \textbf{partiellen Ableitungen
          nach allen Gewichten} gebildet werden.

    \item Bei vielen Schichten und Gewichten wird dieser Weg schnell
          \textbf{unübersichtlich}.
\end{itemize}

\column{0.48\textwidth}
\footnotesize
\begin{blockC3}{\footnotesize Einsetzen in die Loss Function}
\[
\begin{aligned}
L
&=
\sum_{(x_i,y_i)\in D}
\Bigl(
W^{(3)}
\operatorname{sig}\bigl(
W^{(2)}
\operatorname{sig}(W^{(1)}x_i)
\bigr)
\\
&\hspace{2.2cm}
- y_i
\Bigr)^2
\end{aligned}
\]
\end{blockC3}

\vspace{0.3cm}

\begin{blockC4}{\footnotesize Gesuchte Ableitungen}
\[
    \frac{\partial L}{\partial w_{1,1}^{(1)}},
    \frac{\partial L}{\partial w_{1,2}^{(1)}},
    \dots,
    \frac{\partial L}{\partial w_{1,4}^{(3)}},
\]
\end{blockC4}

\end{columns}

\end{frame}

\begin{frame}{Problem: Verkettete Funktionen}

\begin{columns}
\column{0.5\textwidth}

\begin{itemize}
    \item Die Netzfunktion besteht aus vielen \textbf{verketteten Funktionen}.

    \item Jede Schicht nutzt die Ausgabe der vorherigen Schicht als Eingabe.

    \item Deshalb reicht eine einfache Ableitung nicht aus.

    \item Stattdessen muss die \textbf{Kettenregel} mehrfach angewendet werden.

    \item Dabei unterscheidet man jeweils zwischen äußerer und innerer
          Ableitung.
\end{itemize}

\column{0.45\textwidth}

\begin{blockC1}{Verkettung im Netz}
\[
\mathbf{x}
\rightarrow
W^{(1)}\mathbf{x}
\rightarrow
\operatorname{sig}(W^{(1)}\mathbf{x})
\]
\[
\rightarrow W^{(2)}(\cdot)\mathbf{x}
\rightarrow
\dots
\rightarrow
\hat{y}
\]
\end{blockC1}

\vspace{0.3cm}

\begin{blockC1}{Kettenregel}
\[
    \frac{\partial L}{\partial w}
    =
    \frac{\partial L}{\partial \hat{y}}
    \cdot
    \frac{\partial \hat{y}}{\partial w}
\]
\[
\text{äußere Ableitung}
\cdot
\text{innere Ableitung}
\]
\end{blockC1}

\end{columns}

\source{\cite{Frochte2020}}
\end{frame}

\begin{frame}{Idee der Backpropagation}
\footnotesize
\begin{columns}
\column{0.52\textwidth}

\begin{itemize}
    \item \textbf{Backpropagation} berechnet die vielen Anwendungen der
          Kettenregel systematisch.

    \item Zuerst wird die Eingabe vorwärts durch das Netz geleitet.

    \item Danach wird der Fehler \textbf{rückwärts durch das Netz propagiert}.

    \item Für jede Schicht wird bestimmt, welchen \textbf{Anteil ihre Gewichte} am
          \textbf{Gesamtfehler} haben.

    \item Dadurch erhält man effizient alle Ableitungen für den
          Gradientenabstieg.
\end{itemize}

\column{0.45\textwidth}

\begin{block}{\footnotesize Vorwärtsrechnung}
\[
    \mathbf{x}
    \longrightarrow
    \hat{y}
    \longrightarrow
    L
\]
\end{block}

\vspace{0.3cm}

\begin{blockC2}{\footnotesize Rückwärtsrechnung}
\[
    L
    \longrightarrow
    W^{(3)}
    \longrightarrow
    W^{(2)}
    \longrightarrow
    W^{(1)}
\]
\end{blockC2}

\vspace{0.3cm}

\begin{block}{\footnotesize Ergebnis}
\[
    \frac{\partial L}{\partial W^{(1)}},
    \quad
    \frac{\partial L}{\partial W^{(2)}},
    \quad
    \frac{\partial L}{\partial W^{(3)}}
\]
\end{block}

\end{columns}

\end{frame}

\begin{frame}{Backpropagation als effiziente Kettenregel}
\footnotesize
\begin{columns}
\column{0.6\textwidth}

\begin{itemize}
      
    \item Der direkte Weg wäre:
          \textbf{einsetzen, ausmultiplizieren, nach allen Gewichten ableiten}.

    \item Das ist bei tiefen Netzen sehr aufwendig.

    \item Backpropagation nutzt die Struktur des Netzes aus.

    \item \textbf{Zwischenergebnisse} aus der Vorwärtsrechnung werden gespeichert.

    \item In der \textbf{Rückwärtsrechnung} werden daraus \textbf{effizient die Ableitungen
          berechnet}.
\end{itemize}

\column{0.35\textwidth}

\begin{blockC1}{\footnotesize Kernidee}
Nicht jede Ableitung wird vollständig neu berechnet.

\vspace{0.3cm}

Stattdessen werden \textbf{Fehleranteile schichtweise} zurückgegeben.
\end{blockC1}

\vspace{0.3cm}

\begin{block}{\footnotesize Für den Gradientenabstieg}
\[
    W^{(k)}
    \leftarrow
    W^{(k)}
    -
    \alpha
    \frac{\partial L}{\partial W^{(k)}}
\]
\end{block}
\end{columns}
\end{frame}

\begin{itemize}
    \item Backpropagation berechnet den \textbf{Einfluss jedes Neurons}
          auf den Gesamtfehler.

    \item Dazu werden Zwischenergebnisse der nächsten Schicht als
          \textbf{Fehlersignal} zurückgegeben.

    \item Die Veränderung des Gewichts von Neuron $j$ zu Neuron $i$ lautet:
          \[
              \Delta w_{i,j}
              =
              -\alpha
              \frac{\partial L}{\partial w_{i,j}}
              =
              -\alpha\,\delta_i\cdot o_j
          \]

    \item $\delta_i$ ist das \textbf{Fehlersignal} des Neurons $i$.

    \item $o_j$ ist die \textbf{Ist-Ausgabe} des Neurons $j$.

    \item $\alpha$ ist die \textbf{Lernrate} des Gradientenabstiegs.
\end{itemize}

\begin{frame}{Backpropagation: Berechnung der Fehlersignale}
\footnotesize
\begin{columns}

\column{0.4\textwidth}

\begin{itemize}
    \item Bei der Berechnung des Fehlersignals $\delta_j$ unterscheidet man,
          ob $j$ ein \textbf{Neuron der Ausgabeschicht} ist oder nicht.

    \item Für ein Ausgabeneuron kann der Fehler direkt mit dem Soll-Wert $y$
          verglichen werden.

    \item Für ein verstecktes Neuron wird das Fehlersignal aus der
          \textbf{nachfolgenden Schicht} zurückgerechnet.

    \item Genau diese rückwärts gerichtete Berechnung ist die Grundidee der
          \textbf{Backpropagation}.
\end{itemize}

\column{0.60\textwidth}

\begin{block}{\footnotesize Fehlersignal}

\noindent\makebox[\linewidth][c]{%
$\displaystyle
\delta_j =
\begin{cases}
\varphi'(\mathrm{net}_j)\,(o_j-y),
& \text{falls } j \text{ Ausgabeneuron ist}, \\[0.4cm]
\varphi'(\mathrm{net}_j)
\displaystyle\sum_k \delta_k\,w_{k,j},
& \text{sonst.}
\end{cases}
$%
}

\end{block}

\begin{blockC2}{\footnotesize Bedeutung}
\begin{itemize}
    \item $\delta_j$: Fehlersignal von Neuron $j$
    \item $o_j$: Ausgabe von Neuron $j$
    \item $y$: gewünschte Ausgabe
    \item $\mathrm{net}_j$: gewichtete Summe der Eingänge von Neuron $j$: $\mathrm{net}_j = \sum_i w_{j,i}\,o_i$
              
    \item $\varphi'(\mathrm{net}_j)$: Ableitung der Aktivierungsfunktion
    \item $w_{k,j}$: Gewicht von Neuron $j$ zu Neuron $k$
\end{itemize}
\end{blockC2}
\end{columns}
\end{frame}


\begin{frame}{Zusammenspiel: Backpropagation \& Gradient Descent}
\par\medskip
\begin{columns}
\column{0.47\textwidth}
\textbf{Backpropagation}
\begin{itemize}
\item Effiziente Berechnung der Gradienten
\item Nutzung der Kettenregel
\item Rückwärts durch das Netzwerk
\item Liefert: $\frac{\partial L}{\partial w}$
\end{itemize}

\column{0.48\textwidth}
\textbf{Gradient Descent}
\begin{itemize}
\item Verwendet Gradienten zur Optimierung
\item Aktualisiert Gewichte iterativ
\item Ziel: Minimierung der Loss Function
\item Update: $w := w - \alpha \cdot \nabla L$
\end{itemize}
\end{columns}

\vspace{1em}
\begin{block}{Merksatz}
Backpropagation liefert die Gradienten. Gradient Descent nutzt sie, um das Netz zu trainieren.
\end{block}
\end{frame}

\begin{frame}{Backpropagation: Lösung 1 -- Forward Propagation}
\footnotesize

\begin{columns}

\column{0.48\textwidth}

\textbf{Verstecktes Neuron}

\[
\begin{aligned}
\mathrm{net}_h
&=
w_{h,1}x_1+w_{h,2}x_2 \\[0.1cm]
&=
0{,}5\cdot 1 + (-0{,}5)\cdot 2 \\[0.1cm]
&=
0{,}5 - 1 \\[0.1cm]
&=
-0{,}5
\end{aligned}
\]

Da $\varphi(x)=x$ gilt:
\[
  o_h=\varphi(\mathrm{net}_h)=-0{,}5.
\]

\column{0.48\textwidth}

\textbf{Ausgabeneuron}

\[
\begin{aligned}
\mathrm{net}_o
&=
w_{o,h}o_h \\[0.1cm]
&=
2\cdot(-0{,}5) \\[0.1cm]
&=
-1
\end{aligned}
\]

Damit ist:
\[
  \hat{y}=o_o=-1.
\]

Der quadratische Fehler beträgt:
\[
  L=(\hat{y}-y)^2=(-1-1)^2=4.
\]

\end{columns}

\end{frame}


%=======================================================================
% Frame 3: Lösung 2 - Backpropagation / Fehlersignale
%=======================================================================
\begin{frame}{Backpropagation: Lösung 2 -- Fehlersignale}
\footnotesize

\begin{columns}

\column{0.48\textwidth}

\textbf{Ausgabeneuron}

Für das Ausgabeneuron gilt:
\[
  \delta_o
  =
  \varphi'(\mathrm{net}_o)(o_o-y)
\]

Einsetzen liefert:
\[
\begin{aligned}
\delta_o
&=
1\cdot(-1-1) \\[0.1cm]
&=
-2.
\end{aligned}
\]

Das Ausgabeneuron trägt also mit dem Fehlersignal
\[
  \delta_o=-2
\]
zum Gesamtfehler bei.

\column{0.48\textwidth}

\textbf{Verstecktes Neuron}

Für ein verstecktes Neuron wird das Fehlersignal aus der nächsten Schicht
zurückgerechnet:
\[
  \delta_h
  =
  \varphi'(\mathrm{net}_h)
  \sum_k \delta_k w_{k,h}.
\]

Da es hier nur ein Ausgabeneuron gibt:
\[
\begin{aligned}
\delta_h
&=
1\cdot \delta_o w_{o,h} \\[0.1cm]
&=
(-2)\cdot 2 \\[0.1cm]
&=
-4.
\end{aligned}
\]

Also:
\[
  \delta_h=-4.
\]

\end{columns}

\end{frame}

\begin{frame}{Backpropagation: Lösung 3 -- Gewichtsänderungen}
\footnotesize

\begin{columns}

\column{0.48\textwidth}

\textbf{Regel}

Die Gewichtsänderung lautet:
\[
  \Delta w_{i,j}
  =
  -\alpha\,\delta_i\,o_j.
\]

Für das Gewicht vom versteckten Neuron zum Ausgabeneuron:
\[
\begin{aligned}
\Delta w_{o,h}
&=
-\alpha\delta_o o_h \\[0.1cm]
&=
-0{,}1\cdot(-2)\cdot(-0{,}5) \\[0.1cm]
&=
-0{,}1.
\end{aligned}
\]

Damit:
\[
  w_{o,h}^{\,neu}
  =
  2 + (-0{,}1)
  =
  1{,}9.
\]

\column{0.48\textwidth}

\textbf{Gewichte zur versteckten Schicht}

Für $w_{h,1}$:
\[
\begin{aligned}
\Delta w_{h,1}
&=
-\alpha\delta_h x_1 \\[0.1cm]
&=
-0{,}1\cdot(-4)\cdot 1 \\[0.1cm]
&=
0{,}4.
\end{aligned}
\]

Für $w_{h,2}$:
\[
\begin{aligned}
\Delta w_{h,2}
&=
-\alpha\delta_h x_2 \\[0.1cm]
&=
-0{,}1\cdot(-4)\cdot 2 \\[0.1cm]
&=
0{,}8.
\end{aligned}
\]

Damit:
\[
  w_{h,1}^{\,neu}=0{,}5+0{,}4=0{,}9
\]
\[
  w_{h,2}^{\,neu}=-0{,}5+0{,}8=0{,}3
\]

\end{columns}

\end{frame}

\begin{frame}{Regularisierung im Multi-Layer-Perceptron}
\footnotesize
\begin{columns}
\column{0.75\textwidth}
\begin{itemize}
  \item MLPs tendieren bei hoher Modellkomplexität zur \textbf{Überanpassung}.
    \item Zur Regularisierung wird der Loss-Funktion häufig ein 
    $L^2$-\textbf{Strafterm} hinzugefügt, der große Gewichte bestraft.
    \item \textbf{Grundgedanke:} Große Gewichte führen dazu, dass das Netz stark auf kleine Änderungen der Eingangsgrößen reagiert.
    Dadurch entstehen steile und empfindliche Entscheidungsfunktionen.
\item Der Parameter  $\alpha$ steuert den Einfluss des $L^2$-\textbf{Strafterms}:
\[
C(\mathbf{W})=\underbrace{L(\mathbf{W})}_{\text{Fehler}}+\alpha\underbrace{\textstyle\sum_i w_i^2}_{\text{Strafterm}}
\]
    \item \textbf{Großes $\alpha$:} kleinere Gewichte, \textbf{glattere} Entscheidungsgrenze,
    Overfitting.

    \item \textbf{Kleines $\alpha$:} größere Gewichte, \textbf{komplexere} Entscheidungsgrenze,
    Unterfitting.
\end{itemize}

\begin{frame}[fragile]{MLP: Daten skalieren}
\begin{columns}
\column{0.55\textwidth}
Ein \textbf{MLP} arbeitet mit \textbf{skalierten} Daten besser. Der
\texttt{StandardScaler} normiert die Merkmale auf Mittelwert $0$ und
Varianz $1$:
\begin{minted}[fontsize=\footnotesize]{python}
from sklearn.preprocessing import StandardScaler
scaler = StandardScaler()
scaler.fit(X_train)
X_train = scaler.transform(X_train)
X_test = scaler.transform(X_test)
\end{minted}

\column{0.40\textwidth}
\begin{blockC1}{Erinnerung: Warum skalieren?}
\begin{itemize}
  \item Merkmale mit großen Werten würden das Training sonst
        \textbf{dominieren}.
  \item Der Scaler wird nur auf den \textbf{Trainingsdaten} gefittet.
  \item Test- und Trainingsdaten werden \textbf{gleich} transformiert.
\end{itemize}
\end{blockC1}
\end{columns}
\source{Multilayer-Perceptron.ipynb}
\end{frame}

\begin{frame}[fragile]{MLP: Netz aufsetzen und trainieren}
\begin{columns}
\column{0.6\textwidth}
\begin{minted}[fontsize=\footnotesize]{python}
from sklearn.neural_network import MLPClassifier
model = MLPClassifier(
      solver='lbfgs',
      max_iter=20000, alpha=1,
      hidden_layer_sizes=(30,20,10),
      random_state=0)
model.fit(X_train, y_train)
print(model.score(X_test, y_test))
\end{minted}

\column{0.4\textwidth}
\footnotesize
\begin{blockC1}{\footnotesize Parameter des MLP}
\begin{itemize}
  \item \texttt{solver}: \textbf{Optimierungsverfahren} für die Gewichte. \texttt{'sgd'} wäre das klassische \emph{Stochastic Graadiend Descent}
  \item \texttt{max\_iter}: maximale \textbf{Iterationszahl}.
  \item \texttt{alpha}: Stärke der \textbf{Regularisierung}.
  \item \texttt{hidden\_layer\_sizes}: \textbf{Größe} der Hidden-Layer, hier
        $30$, $20$, $10$.
  \item \texttt{random\_state}: feste \textbf{Initialisierung}.
\end{itemize}
\end{blockC1}
\end{columns}
\source{Multilayer-Perceptron.ipynb}
\end{frame}
