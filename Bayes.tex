\begin{frame}{Naive-Bayes-Klassifikator: Grundidee}
\begin{columns}
\column{0.58\textwidth}
Der \textbf{Naive-Bayes-Klassifikator} ist ein einfaches, aber oft
erstaunlich leistungsfähiges Verfahren auf Basis der
\textbf{Wahrscheinlichkeitsrechnung}.

\vspace{0.4em}
\begin{itemize}
  \item Grundlage ist der \textbf{Satz von Bayes}.
  \item Gesucht: die \textbf{wahrscheinlichste Klasse} für einen gegebenen
        Merkmalsvektor.
  \item \textbf{„Naiv":} alle Merkmale werden als \textbf{voneinander
        unabhängig} angenommen.
\end{itemize}

\vspace{0.4em}
{\footnotesize\itshape Diese Vereinfachung stimmt selten exakt, in der
Praxis funktioniert sie aber oft sehr gut.}

\column{0.38\textwidth}
\begin{blockC1}{Satz von Bayes}
\centering
\vspace{0.2em}
\[
  P(K \mid M) = \frac{P(M \mid K)\, P(K)}{P(M)}
\]
\vspace{0.2em}

{\footnotesize $K$: Klasse, $M$: Merkmale}
\end{blockC1}
\end{columns}
\end{frame}


\begin{frame}{Satz von Bayes: anschaulich am Beispiel}
\begin{columns}
\column{0.52\textwidth}
\footnotesize
\textbf{100 Studierende.} 60 lernen, 40 nicht. Wer lernt, besteht zu 90\,\%,
wer nicht lernt zu 30\,\%.

\column{0.45\textwidth}
\footnotesize
\textbf{Frage:} Jemand hat \textbf{bestanden}. Wie wahrscheinlich hat die
Person \textbf{gelernt}?

\vspace{0.4em}
Bestanden haben $54 + 12 = 66$ Personen. Davon haben 54 gelernt:
\[
  P(L \mid B) = \frac{54}{66} \approx 0{,}82
\]

\vspace{0.3em}
\begin{blockC1}{Das ist der Satz von Bayes}
\[
  P(L \mid B) = \frac{P(B \mid L)\,P(L)}{P(B)}
              = \frac{0{,}9 \cdot 0{,}6}{0{,}66}
\]
\end{blockC1}

\begin{frame}{Satz der totalen Wahrscheinlichkeit}
\begin{columns}
\column{0.52\textwidth}
\footnotesize
Im Bayes-Nenner steht $P(B)$, also die Wahrscheinlichkeit zu \textbf{bestehen},
ganz ohne Vorbedingung. Wie kommt man an diesen Wert?

\vspace{0.4em}
Bestehen kann man auf \textbf{zwei Wegen}: als Lernende(r) oder als
Nicht-Lernende(r). Beide Wege werden \textbf{aufsummiert}:
\[
  P(B) = \underbrace{P(B \mid L)\,P(L)}_{\text{Pfad „lernen"}}
       + \underbrace{P(B \mid \overline{L})\,P(\overline{L})}_{\text{Pfad „nicht lernen"}}
\]

\vspace{0.3em}
Mit Zahlen:
\[
  P(B) = 0{,}9 \cdot 0{,}6 + 0{,}3 \cdot 0{,}4
       = 0{,}54 + 0{,}12 = 0{,}66
\]

\begin{frame}{Naive Bayes: Grundidee}
Der \textbf{Satz von Bayes} besagt:
\[
  P(A \mid B) = \frac{P(A)\cdot P(B \mid A)}{P(B)}
\]

Unsere Beobachtung ist $B = (\text{Helligkeit}=7,\ \text{Länge}=10)$. Damit
berechnen wir die Klassenzugehörigkeit:

\begin{columns}[T,onlytextwidth]
\begin{column}{0.47\textwidth}
\begin{blockC1}{Lachs}
\footnotesize
\[
  P(\text{Lachs} \mid B) =
  \frac{P(\text{Lachs})\cdot P(B \mid \text{Lachs})}{P(B)}
\]
\end{blockC1}
\end{column}

\begin{column}{0.48\textwidth}
\begin{blockC2}{Barsch}
\footnotesize
\[
  P(\text{Barsch} \mid B) =
  \frac{P(\text{Barsch})\cdot P(B \mid \text{Barsch})}{P(B)}
\]
\end{blockC2}
\end{column}
\end{columns}

\vspace{0.4em}
{\itshape Der Nenner $P(B)$ ist für beide Klassen gleich. Für den
Vergleich genügt der Zähler.}
\source{\cite{duda2001pattern}}
\end{frame}


\begin{frame}{Naive Bayes: Vom Bayes-Satz zur Rechenregel}
\textbf{Schritt 1:} Für den Vergleich der Klassen genügt der
\textbf{Zähler}, da der Nenner $P(B)$ für beide gleich ist:
\[
  P(K \mid B) \ \propto\ P(K)\cdot P(B \mid K)
\]

\vspace{0.6em}
\textbf{Schritt 2:} Mit der Beobachtung
$B = (\text{Helligkeit}=7,\ \text{Länge}=10)$ brauchen wir also
\[
  P(K)\cdot P(\text{Helligkeit}=7,\ \text{Länge}=10 \mid K)
\]
für jede Klasse $K \in \{\text{Lachs}, \text{Barsch}\}$.
\source{\cite{duda2001pattern}}
\note{$\propto$ bedeutet Proportionalität}
\end{frame}

\begin{frame}{Naive Bayes: die „naive" Annahme}
\textbf{Schritt 3:} Die Wahrscheinlichkeit
$P(\text{H.}=7,\ \text{L.}=10 \mid \text{Lachs})$ lässt sich nicht einfach
berechnen, da sie von \textbf{zwei} Zufallsvariablen abhängt.

\vspace{0.4em}
Der Naive-Bayes-Klassifikator nimmt (naiv) an, dass alle Kenngrößen
\textbf{voneinander unabhängig} sind. Dann gilt:
\[
  P(\text{H.}{=}7,\ \text{L.}{=}10 \mid \text{Lachs}) =
  P(\text{H.}{=}7 \mid \text{Lachs})\cdot P(\text{L.}{=}10 \mid \text{Lachs})
\]

\vspace{0.4em}
\begin{columns}[T,onlytextwidth]
\begin{column}{0.54\textwidth}
Aus dem schwer schätzbaren gemeinsamen Term wird so ein \textbf{Produkt}
einzelner Wahrscheinlichkeiten -- jede pro Merkmal getrennt schätzbar.
\end{column}

\begin{column}{0.42\textwidth}
\begin{blockC1}{Entscheidungsregel}
Wähle die Klasse $K$, die
\[
  P(K)\cdot \prod_i P(M_i \mid K)
\]
maximiert.
\end{blockC1}
\end{column}
\end{columns}
\source{\cite{duda2001pattern}}
\end{frame}


\begin{frame}{Naive Bayes: Schätzung der Wahrscheinlichkeiten}
\begin{columns}
\column{0.52\textwidth}
Die einzelnen Wahrscheinlichkeiten wie $P(\text{H.}{=}7 \mid \text{Lachs})$
schätzt man aus den \textbf{Stichproben} ab.

\vspace{0.4em}
\begin{itemize}
  \item Pro Merkmal und Klasse wird z.~B. eine \textbf{Normalverteilung}
        angenommen.
  \item Aus den Trainingsdaten bestimmt man \textbf{Mittelwert} $\mu$ und
        \textbf{Varianz} $\sigma^2$.
  \item Der Wahrscheinlichkeitswert ergibt sich dann aus der
        \textbf{Dichtefunktion}.
\end{itemize}

\vspace{0.4em}
{\footnotesize\itshape Jedes Merkmal jeder Klasse erhält so seine eigene
Gauß-Kurve.}

\begin{frame}{Naive Bayes: Die zwei naiven Annahmen}
\begin{columns}
\column{0.62\textwidth}
Der Naive-Bayes-Klassifikator trifft zwei \textbf{vereinfachende Annahmen}:

\vspace{0.4em}
\begin{enumerate}[1.]
  \item Alle Kenngrößen sind \textbf{stochastisch unabhängig}. Der Wert
        einer Kenngröße hängt nicht von einer anderen ab.
        \begin{itemize}
          \item Für \textbf{Länge} und \textbf{Helligkeit} ist das vermutlich
                korrekt.
          \item Für \textbf{Länge} und \textbf{Gewicht} aber nicht. Mit der
                Länge steigt meist auch das Gewicht.
        \end{itemize}
  \item Die Werte lassen sich durch eine \textbf{Verteilung} beschreiben
        (z.\,B. Normalverteilung).
\end{enumerate}

\vspace{0.4em}
{\footnotesize\itshape Beide Annahmen sind etwas „naiv" $\rightarrow$ daher der Name.}

\column{0.32\textwidth}
\centering

\begin{frame}{Naive Bayes: Vor- und Nachteile}
\begin{columns}[T]
\column{0.50\textwidth}
\begin{blockC1}{Vorteile}
\begin{itemize}
  \item Einfaches und \textbf{sehr schnelles} Modell.
  \item Funktioniert gut mit \textbf{wenigen} Trainingsdaten.
  \item Kommt sehr gut mit \textbf{hochdimensionalen} Eingabedaten zurecht.
  \item Gut für \textbf{Mehrklassen}-Probleme geeignet.
\end{itemize}
\end{blockC1}

\column{0.46\textwidth}
\begin{blockC2}{Nachteile}
\begin{itemize}
  \item Die vereinfachenden \textbf{Grundannahmen} sind oft nicht gegeben.
  \item Bei \textbf{diskreten} Merkmalswerten und kleinen Stichproben droht Wahrscheinlichkeit
        $0$.
\end{itemize}
\end{blockC2}
\end{columns}

\vspace{0.6em}
\centering
{\footnotesize\itshape Schnell und robust bei wenig Daten -- aber durch die
Unabhängigkeitsannahme begrenzt.}
\source{\cite{duda2001pattern}}
\end{frame}
