werden.

\vspace{0.4em}
\textbf{Wann ist ein Baum „gut"?} Ein guter Entscheidungsbaum
\begin{itemize}
  \item macht \textbf{möglichst wenige Fehler} und
  \item ist  unter den fehlerärmsten Bäumen  der \textbf{kleinste}, der
        eine gültige Entscheidung liefert.
\end{itemize}

\vspace{0.4em}
{\footnotesize\itshape Größenmaße: Baumhöhe, Anzahl der Blätter oder
Pfadlängensumme.}
\end{columns}

\begin{frame}{Wie findet man einen kleinen, genauen Baum?}

\begin{itemize}
  \item Problem ist \textbf{NP-vollständig}.
  \item Rechnerisch herausfordernd.
  \item Liefert potenziell die \textbf{genaueste} Lösung.
\end{itemize}
\end{blockC1}

\column{0.48\textwidth}
\begin{blockC2}{Heuristiken verwenden}
\begin{itemize}
  \item Liefert \textbf{praktikable}, aber nicht perfekte Lösungen.
  \item Besonders wichtig bei \textbf{größeren Datenmengen}.
\end{itemize}
\end{blockC2}
\end{columns}

\begin{frame}{ID3-Algorithmus}
\begin{columns}
\column{0.65\textwidth}
\textbf{ID3} (Iterative Dichotomiser 3)
\begin{itemize}
  \item Entwickelt von \textbf{Ross Quinlan} in den 1980er Jahren.
  \item Einfache Struktur $\rightarrow$ ideal als \textbf{Einstiegsbeispiel}.
  \item Fokus: Klassifikation mit \textbf{nominalen Merkmalen}.
\end{itemize}

\vspace{0.4em}
\begin{blockC1}{Eigenschaften}
\begin{itemize}
  \item Effizient bei vielen Features.
  \item Erzeugt vergleichsweise \textbf{einfache} Bäume.
  \item \textbf{Keine Garantie} für optimale Bäume.
\end{itemize}
\end{blockC1}

\column{0.3\textwidth}
\begin{blockDef}{Dichotomiser}
Verfahren zur \textbf{schrittweisen Aufspaltung} von Daten in Teilgruppen

\begin{frame}{Algorithmus zur Entscheidungsbaumerzeugung}
\begin{columns}
\column{0.6\textwidth}
\scriptsize
\begin{algorithm}[H]
\SetAlgoCaptionLayout{scriptsize}
\KwIn{Beispielmenge $D$, Merkmale $A = \{A_1, \dots, A_n\}$}
\KwOut{Entscheidungsbaum}
\SetKwFunction{FMain}{GENTREE}
\SetKwProg{Fn}{Prozedur}{:}{}
\Fn{\FMain{$D$, $A$}}{
  \If{$D$ einklassig}{
    Blatt mit dieser Klasse\;
    \Return\;
  }
  \If{$A = \emptyset$}{
    Blatt mit Mehrheitsklasse\;
    \Return\;
  }
  Wähle $A_i \in A$ mit $r$ Werten\;
  Teile $D$ nach $A_i$ in $D_1, \dots, D_r$\;
  \ForEach{$D_j$}{
    \eIf{$D_j \neq \emptyset$}{
      \FMain{$D_j$, $A \setminus \{A_i\}$}\;
    }{
      Blatt mit Mehrheitsklasse\;
    }
  }
}
\caption{Entscheidungsbaum-Erzeugung}
\end{algorithm}

\column{0.35\textwidth}
\begin{itemize}
  \item Die konketen Verfahren unterscheiden sich nur in der \textbf{Merkmalsauswahl}.
  \item Der Grundalgorithmus erlaubt auch Prüfungen mit \textbf{mehr als zwei
        Ausgängen}.
  \item Wir arbeiten aber stets mit \textbf{Binärbäumen}.
  \item \textbf{Rekursiver Aufruf} für alle Teilmengen.
\end{itemize}
\end{columns}
\source{\cite{Frochte2020}}
\end{frame}

\begin{frame}{Wahrscheinlichkeit und Informationsgehalt}

\begin{columns}[T]

\begin{column}{0.6\textwidth}

\begin{blockDef}{Alphabet}
\(\{s_i\}\) mit \(i = 1, 2, \dots, k\) sei die Menge aller unterschiedlichen,

\end{blockDef}

und Symbolfolgen.
\end{block}

\end{column}

\begin{column}{0.35\textwidth}

\textbf{Bezeichnungen}

\begin{itemize}
  \item \(\{s_i\}\): Alphabet mit \(k\) Symbolen
  \item \(s_i\): einzelnes Symbol
  \item Wahrscheinlichkeiten: Basis für Entropie und Information
\end{itemize}

\end{column}

\end{columns}
\source{\cite{strutz2025bilddatenkompression}}
\end{frame}

\begin{frame}{Auftretenswahrscheinlichkeit von Symbolen}

\begin{blockDef}{Auftretenswahrscheinlichkeit}
Jedes Symbol \(s_i\) besitzt eine \textbf{Auftretenswahrscheinlichkeit} \(p_i\) mit
\[
\sum_{i=1}^{k} p_i = 1.
\]
In der Praxis werden die \(p_i\) oft durch die relative Häufigkeit geschätzt:
\[
p_i \approx \frac{h(s_i)}{n},
\]
wobei \(h(s_i)\) die absolute Häufigkeit von \(s_i\) und \(n\) die Gesamtanzahl
der Symbole in der Sequenz ist.
\end{blockDef}

\source{\cite{strutz2025bilddatenkompression}}
\end{frame}

\begin{frame}{Informationsgehalt eines Symbols}

\begin{columns}

\begin{column}{0.35\textwidth}

\begin{blockDef}{Informationsgehalt}
Der \textbf{Informationsgehalt} eines Symbols \(s_i\) mit Wahrscheinlichkeit \(p_i\) ist definiert als
\[
I(s_i) = \log_2 \frac{1}{p_i} \quad [\text{bit}].
\]
\end{blockDef}

\end{column}

\begin{column}{0.6\textwidth}

\begin{itemize}
  \item \textbf{Häufige (wahrscheinliche) Symbole} liefern \textbf{wenig Information}.
  \item \textbf{Seltene (überraschende) Symbole} liefern \textbf{viel Information}.
\end{itemize}

\medskip

Die Einheit des Informationsgehalts ist \(\mathbf{[\text{bit}]}\).  
\textbf{„Bit“} (binary digit) bezeichnet dagegen ein \textbf{zählbares 0/1-Element}, z.~B.:
Ein Byte besteht aus \textbf{8 Bits}.

\end{column}

\end{columns}

\source{\cite{strutz2025bilddatenkompression}}
\end{frame}

\begin{frame}{Quelle mit unabhängigen Symbolen}
  \small
\begin{columns}[T]
  \begin{column}{0.47\textwidth}
    \textbf{Ausgangssituation}

    \begin{itemize}
      \item Nachrichten bestehen aus \textbf{Folgen} von Symbolen
            (z.~B. Buchstaben, Messwerte, Pixelwerte).
      \item Alphabet mit \(k\) Symbolen \(s_1,\dots,s_k\), Wahrscheinlichkeiten \(p_i = P(s_i)\).
      \item Gesucht: \textbf{mittlerer Informationsgehalt} pro Symbol.
    \end{itemize}

    \vspace{0.3cm}
    \textbf{Informationsgehalt von Einzelinformation}
    \[
      I(s_i) = \log_2\!\left(\frac{1}{p_i}\right)
    \]
  \end{column}

  \begin{column}{0.48\textwidth}
    \begin{blockDef}{\small Statistisch unabhängige Symbole}
      \begin{itemize}
        \item Symbole sind \textbf{statistisch unabhängig}, wenn das Auftreten eines Symbols
              die Wahrscheinlichkeiten der nächsten Symbole \textbf{nicht verändert}.
        \item \textbf{Beispiel unabhängig:}
              wiederholtes Werfen einer fairen Münze.
        \item \textbf{Beispiel nicht unabhängig:}
              Buchstaben in natürlicher Sprache (z.~B. nach „Q“ folgt fast immer „U“).
        \item Modellannahme: Die \(p_i\) bleiben von Symbol zu Symbol \textbf{konstant}.
      \end{itemize}
    \end{blockDef}
  \end{column}
\end{columns}
\source{\cite{strutz2025bilddatenkompression}}
\end{frame}

\begin{frame}{Signalentropie}
\small
\begin{columns}]
  \begin{column}{0.66\textwidth}

    \begin{blockDef}{\small Entropie}
    Für eine Quelle mit \(k\) Symbolen \(s_i\) und Wahrscheinlichkeiten \(p_i\) ist die \textbf{Entropie} \(H\) definiert durch:
    \[
      H
      = \sum_{i=1}^{k} p_i \cdot I(s_i)
      = \sum_{i=1}^{k} p_i \cdot \log_2\!\left(\frac{1}{p_i}\right)
      = - \sum_{i=1}^{k} p_i \cdot \log_2(p_i)
    \]

    Sie hat die Einheit [\text{bit/Symbol}].
    \end{blockDef}
  \end{column}

  \begin{column}{0.3\textwidth}
    \textbf{Interpretation}

    \begin{itemize}
      \item \(H\) ist der \textbf{mittlere Informationsgehalt} pro Symbol.
      \item Je \textbf{unsicherer} die Quelle (d.\,h. je gleichverteilter die \(p_i\)),
            desto größer ist \(H\).
      \item \(H\) ist die theoretische Untergrenze für die mittlere Anzahl
            von Bits pro Symbol bei optimaler Kodierung.
    \end{itemize}
  \end{column}
\end{columns}
\source{\cite{strutz2025bilddatenkompression}}
\end{frame}

\begin{frame}{Entropie: Sichere Quelle}
\begin{columns}
  \begin{column}{0.7\textwidth}
    \textbf{Extremfall 1: Sichere Quelle}
    \begin{itemize}
      \item Nur \textbf{ein} Symbol tritt auf: 
            \(
              p_1 = 1,\quad p_i = 0 \;\text{für } i = 2,\dots,k.
            \)
      \item Keine Unsicherheit, keine Überraschung.
    \end{itemize}
\medskip
    \textbf{Berechnung der Entropie}
    \[
      H = - \sum_{i=1}^{k} p_i \log_2(p_i)
         = - \bigl(1 \cdot \log_2(1) + 0 \cdot \log_2(0) + \dots \bigr).
    \]
    \[
      \log_2(1) = 0,\quad
      0 \cdot \log_2(0) := 0
      \;\Rightarrow\;
      H = -0 = 0.
    \]
    \[
      \boxed{H = 0 \;\text{bit/Symbol}}
    \]
  \end{column}

\begin{columns}
  \begin{column}{0.7\textwidth}
    \textbf{Extremfall 2: Gleichverteilte Quelle}
    \begin{itemize}
      \item Alle \(k\) Symbole gleich wahrscheinlich:             \(
              \forall i. \; (p_i = \frac{1}{k}) 
            \)
      \item Maximale Unsicherheit, jedes Symbol gleichermaßen „überraschend“.
    \end{itemize}

    \textbf{Berechnung der Entropie}
    \[
      H = - \sum_{i=1}^{k} p_i \log_2(p_i)
        = - \sum_{i=1}^{k} \frac{1}{k} \log_2\!\left(\frac{1}{k}\right).

    \[
      H = - k \cdot \frac{1}{k} \log_2\!\left(\frac{1}{k}\right)
        = - \log_2\!\left(\frac{1}{k}\right)
        = \log_2(k).
    \]

    \[
      \boxed{H_{\max} = \log_2(k) \;\text{bit/Symbol}}

\begin{frame}{Bäume aus Lerndaten erstellen: Bedingte Entropie}
\begin{columns}
\column{0.60\textwidth}
Wir übertragen das Prinzip auf das \textbf{Radfahren}-Beispiel. Was
passiert an einem Knoten?

\vspace{0.4em}
\begin{itemize}
  \item Jedes Attribut (z.\,B. Schnee: ja/nein) ist ein
        \textbf{Zufallsexperiment} $W$.
  \item Das Zielattribut (Rad fahren: ja/nein) ist $V$.
  \item \textbf{Fragestellung:} „Wie unsicher ist $V$, wenn wir $W$ kennen?"
\end{itemize}


\column{0.35\textwidth}
\begin{blockDef}{Bedingte Entropie}
\centering
\vspace{0.2em}
\[
  H(V \mid W)
\]
\vspace{0.2em}
\end{blockDef}

\centering
{\footnotesize\itshape Unsicherheit über $V$ bei bekanntem $W$.}
\end{columns}
\source{\cite{Frochte2020}}
\end{frame}


\begin{frame}{Bedingte Entropie: Formale Definition}
\begin{columns}
\column{0.62\textwidth}
\footnotesize
Sei $V$ ein Zufallsexperiment mit Ausgängen $A_1, \dots, A_k$ und $W$ ein
zweites mit Ausgängen $B_1, \dots, B_s$. Dann gilt:
\[
  H(V \mid W) = \sum_{j=1}^{s} P(B_j)\, H(V \mid B_j),
\]
mit
\[
  H(V \mid B_j) = -\sum_{i=1}^{k} P(A_i \mid B_j)\,\log_2\bigl(P(A_i \mid B_j)\bigr).
\]

Eingesetzt ergibt sich:
\[
  H(V \mid W) = -\sum_{j=1}^{s} P(B_j) \sum_{i=1}^{k}
  P(A_i \mid B_j)\,\log_2\bigl(P(A_i \mid B_j)\bigr).
\]

\column{0.35\textwidth}
\footnotesize
\begin{blockC1}{\footnotesize Anmerkung}
Die \textbf{bedingte Entropie} ist die \textbf{gewichtete mittlere
Entropie} nach der Aufteilung gemäß $W$. Die Gewichtung erfolgt über die
Wahrscheinlichkeiten der einzelnen Ausprägungen von $W$ -- also über die
\textbf{Größe der entstehenden Teilmengen}.

\end{blockC1}

{\footnotesize\itshape Die Wahrscheinlichkeiten werden durch
relative Häufigkeiten aus den Trainingsdaten angenähert.}

\end{columns}
\source{\cite{Frochte2020}}
\end{frame}

\begin{frame}{Bedingte Entropie und Informationsgewinn}
\begin{columns}
\column{0.55\textwidth}
Über die Entropie $H(V)$ und die bedingte Entropie $H(V \mid W)$ können wir den \textbf{Informationsgewinn} berechnen.\\
\medskip
Der Informationsgewinn misst, wie stark die Kenntnis von $W$ die
Unsicherheit über $V$ senkt:
\[
  IG(V, W) = H(V) - H(V \mid W).
\]

\begin{itemize}
  \item $H(V)$: Unsicherheit \textbf{vor} dem Split.
  \item $H(V \mid W)$: Unsicherheit \textbf{nach} dem Split.
\end{itemize}

\column{0.40\textwidth}
\begin{blockC1}{Split-Kriterium}
\centering
\vspace{0.2em}
ID3 wählt an jedem Knoten das Attribut $W$ mit dem
\textbf{größten Informationsgewinn} $IG(V, W)$.
\vspace{0.2em}
\end{blockC1}

\vspace{0.4em}
\begin{blockC2}{Vereinfachung}
$H(V)$ ist an einem Knoten \textbf{konstant}. Es genügt daher, $H(V \mid W)$
zu berechnen und das \textbf{kleinste} zu wählen.
\end{blockC2}
\end{columns}
\source{\cite{Frochte2020}}
\end{frame}

\begin{frame}{Zusammenfassung und Erweiterung}
\begin{columns}
\column{0.47\textwidth}
\begin{blockC1}{ID3-Algorithmus}

\begin{itemize}
  \item Baut Bäume auf \textbf{diskreten} Merkmalen.
  \item Attributwahl über \textbf{maximalen Information Gain}.
  \item Arbeitet \textbf{rekursiv} bis zur reinen Klassifikation.
\end{itemize}
\end{blockC1}

\column{0.48\textwidth}
\begin{blockC3}{Numerische Merkmale}

\textbf{Quantitative} Merkmale brauchen eine Sonderbehandlung:
\begin{itemize}
  \item Binär-Splits an \textbf{Schwellenwerten} (z.\,B. „Druck < 800?").
  \item Sortieren und optimale Trennpunkte suchen.
\end{itemize}
\end{blockC3}
\end{columns}

\vspace{0.4em}

\textbf{Erweiterte Algorithmen}
\begin{itemize}
  \item \textbf{C4.5} -- Nachfolger von ID3; unterstützt numerische Merkmale
        und \textbf{Pruning}.
  \item \textbf{CART} (Classification and Regression Trees) -- Klassifikation
        \textbf{und} Regression.
\end{itemize}
\source{\cite{Frochte2020}}
\end{frame}


\begin{frame}{Entscheidungsbäume: Vor- und Nachteile}
\begin{columns}[T]
\column{0.49\textwidth}
\begin{blockC4}{Vorteile}
\begin{itemize}
  \item Entscheidungen sehr gut \textbf{nachvollziehbar},  der Weg eines
        Datensatzes durch den Baum ist klar verfolgbar.
  \item Berechnungen \textbf{schnell und einfach}.
  \item Baum und Entscheidungen gut \textbf{visualisierbar}.
\end{itemize}
\end{blockC4}

\column{0.49\textwidth}
\begin{blockC2}{Nachteile}
\begin{itemize}
  \item Neigung zur \textbf{Überanpassung} an die Trainingsdaten
        (Overfitting).
  \item Nur \textbf{senkrechte und waagerechte} Trennebenen $\rightarrow$ komplexe
        Entscheidungsgrenzen schwer nachbildbar.
\end{itemize}
\end{blockC2}
\end{columns}

\vspace{0.8em}
\centering
{\footnotesize\itshape Gut interpretierbar und schnell -- aber anfällig für
Overfitting und auf achsenparallele Grenzen beschränkt.}

\begin{frame}[fragile]{Klassifikator speichern und laden}
\begin{columns}
\column{0.50\textwidth}
\footnotesize
\textbf{Speichern}
\begin{minted}[fontsize=\footnotesize]{python}
import pickle

file_name = "modell.pickle"
pickle.dump(model, open(file_name, "wb"))
\end{minted}

\vspace{0.4em}
\textbf{Laden und klassifizieren}
\begin{minted}[fontsize=\footnotesize]{python}
import numpy as np

loaded_model = pickle.load(
    open(file_name, "rb"))
messwerte = np.array([[5.2, 3.4, 1.5, 0.2]])
loaded_model.predict(messwerte)
\end{minted}

\column{0.45\textwidth}
Ein trainiertes Modell lässt sich \textbf{serialisieren} und später wieder
einlesen.

\vspace{0.4em}
\begin{itemize}
  \item \texttt{pickle} eignet sich für einfache Zwecke und
        \textbf{vertrauenswürdige} Quellen.
  \item \texttt{dump} schreibt das Modell, \texttt{load} liest es zurück.
\end{itemize}

\vspace{0.4em}
\begin{blockC2}{\footnotesize Achtung}
\footnotesize
\texttt{pickle}-Dateien aus \textbf{unbekannten} Quellen niemals laden --
sie können beliebigen Code ausführen.
\end{blockC2}
\end{columns}
\source{EntscheidungsbaumIrisMitExport.ipynb}
\end{frame}
