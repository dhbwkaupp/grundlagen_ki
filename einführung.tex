\begin{frame}{Was ist Künstliche Intelligenz?}
\begin{columns}
\column{0.6\linewidth}
\textbf{Definition:}
\begin{itemize}
\item \textbf{Künstliche Intelligenz (KI)} bezeichnet die Fähigkeit von Maschinen, Aufgaben auszuführen, die normalerweise \textbf{menschliche Intelligenz} erfordern.
\item Dazu gehören \textbf{Lernen}, \textbf{Problemlösen}, \textbf{Mustererkennung} und \textbf{Entscheidungsfindung}.
\item KI basiert auf \textbf{Algorithmen} und \textbf{Daten} und wird in vielen Anwendungen eingesetzt, z.\,B. \textbf{Sprachverarbeitung} oder \textbf{Bildanalyse}.
\end{itemize}

\end{columns}
\end{frame}

\begin{frame}{KI-Methoden in der Praxis: Überblick}
  \begin{columns}
    \begin{column}{0.60\linewidth}
      \textbf{Klassifikation} -- typische Fragen:
      \begin{itemize}
        \item Gehört dieses Beispiel zu \textbf{Klasse A, B, C}?
        \item Ist eine Transaktion \textbf{betrügerisch} oder \textbf{legitim}?
        \item Spamnachricht? \textbf{ja}, \textbf{nein}, \textbf{vielleicht}.  
      \end{itemize}

      \vspace{0.5em}

      \textbf{Regression} -- typische Fragen:
      \begin{itemize}
        \item Welchen \textbf{numerischen Wert} nimmt eine Größe an?
        \item Wie hoch wird der \textbf{Preis}, das \textbf{Risiko} oder der \textbf{Verbrauch} sein?
      \end{itemize}
    \end{column}
    \begin{column}{0.35\linewidth}
      \begin{block}{Typische Algorithmen}
        \begin{itemize}
          \item Lineare und logistische Regression
          \item Statistische Modelle (Nayes),
          \item Entscheidungsbäume, Random Forests, 
          \item k-NN, SVM, einfache neuronale Netze
        \end{itemize}
      \end{block}

    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Generatives Modell: Entstehung}
  \begin{columns}
    \begin{column}{0.60\linewidth}
      \begin{itemize}
        \item Ein Computerprogramm analysierte \textbf{15\,000 handgemalte Porträts}
              aus dem 14. bis 20.\ Jahrhundert.
        \item Aus den Bilddaten wurde ein \textbf{Klassifikator (Diskriminator)}
              trainiert, der misst, wie stark ein Bild den Datenbankbildern ähnelt.
        \item Ein \textbf{Generator} erzeugte so lange zufällige Bilder, bis der
              Klassifikator eine \textbf{ausreichend hohe Ähnlichkeit} attestierte.
      \end{itemize}
    \end{column}

\begin{frame}{Signatur: Algorithmus hinter Edmond de Belamy}
  \begin{itemize}
    \item Die \textbf{Signatur} beschreibt den Algorithmus, mit dem das Bild
          erzeugt wurde.
    \item Das Formelzeichen \(\mathbb{E}\) steht für den \textbf{Erwartungswert},
          also den in der Statistik verwendeten Durchschnitt.
    \item Das Verfahren wertet damit \textbf{statistische Merkmale} der
          zugrunde liegenden Bilder aus.
  \end{itemize}

\begin{frame}{Begriffe beim maschinellen Lernen}
  \begin{columns}
    \begin{column}{0.6\linewidth}
      \begin{itemize}
        \item Beim \textbf{überwachten maschinellen Lernen} haben wir eine \textbf{Eingabemenge} \(X\)
              und eine \textbf{Ausgabemenge} \(Y\).
        \item Ein Eingabedatum besteht meist aus mehreren \textbf{Kenngrößen} (Features) und wird als
              \(n\)-Tupel \(\mathbf{x} = (x_1,\dots,x_n) \in X\) dargestellt.
        \item Die Kenngrößen beschreiben gemeinsam ein Objekt und bilden einen \textbf{Eingabevektor}
              in einem \(n\)-dimensionalen Merkmalsraum.
        \item Das \textbf{Modell} ist eine Funktion \(f : X \rightarrow Y\), die jedem Eingang
              \(\mathbf{x} \in X\) einen \textbf{geschätzten Ausgabewert} \(\hat{y} = f(\mathbf{x})\) zuordnet.
      \end{itemize}

      \vfill
      \source{\cite{Frochte2020}}
    \end{column}

\begin{frame}{Bilddaten und Kenngrößen}
  \footnotesize
  \begin{columns}
    \begin{column}{0.5\linewidth}
      \centering
      \includegraphics[width=\linewidth]{figures/fussgaenger.jpg}
    \end{column}
    \begin{column}{0.5\linewidth}
      \begin{itemize}
        \item Im technischen Umfeld sollen oft \textbf{Prognosen aus Bild- oder Signaldaten} getroffen werden.
        \item Hier werden die \textbf{Kenngrößen} oft aus den \textbf{Rohdaten}  berechnet.
        \item Beispiel \textbf{Fahrerassistenzsysteme}: Fußgänger im Fahrschlauch werden per Frontkamera erkannt.
        \item Das ist eine \textbf{Klassifikationsaufgabe} -- nicht das Bild selbst wird klassifiziert.
        \item Das Bild wird gerastert; für jedes Quadrat wird die Richtung des größten Hell-Dunkel-Übergangs
              (Histogram of Oriented Gradients, HOG) als Kenngröße verwendet.
      \end{itemize}

    \end{column}
